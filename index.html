import { serve } from "https://deno.land/std@0.177.0/http/server.ts" import { createClient } from "https://esm.sh/@supabase/supabase-js@2.7.1" import { XMLParser } from "https://esm.sh/fast-xml-parser@4.2.7" const supabaseUrl = Deno.env.get("SUPABASE_URL") as string; const supabaseKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY") as string; const supabase = createClient(supabaseUrl, supabaseKey); serve(async (req) => { const targetBase64 = btoa("admin:Filere2"); const authHeader = req.headers.get("authorization"); const xAuthHeader = req.headers.get("x-authorization"); if (authHeader !== `Basic ${targetBase64}` && xAuthHeader !== targetBase64) { return new Response("Accesso negato: password errata", { status: 401 }); } try { const contentType = req.headers.get("content-type") || ""; if (!contentType.includes("multipart/form-data")) { throw new Error("Richiesta non valida. Previsto multipart/form-data."); } const formData = await req.formData(); const fileInput = formData.get("file"); const fileNameDanea = formData.get("fileName") as string | null; if (!fileInput || !(fileInput instanceof File)) { throw new Error("Nessun file trovato nella richiesta."); } // ================================================================= // FASE A: IMMAGINI // ================================================================= if (fileNameDanea || fileInput.name.toLowerCase().endsWith('.jpg') || fileInput.name.toLowerCase().endsWith('.png')) { let actualFileName = fileNameDanea || fileInput.name; const lastDotIndex = actualFileName.lastIndexOf('.'); let nameWithoutExt = actualFileName; if (lastDotIndex !== -1) nameWithoutExt = actualFileName.substring(0, lastDotIndex); let cleanName = nameWithoutExt.split(' ')[0]; actualFileName = cleanName + '.jpg'; const fileData = new Uint8Array(await fileInput.arrayBuffer()); const { error } = await supabase.storage.from('BIrre').upload(actualFileName, fileData, { upsert: true, contentType: 'image/jpeg' }); if (error) throw error; return new Response("OK", { status: 200, headers: { "Content-Type": "text/plain" } }); } // ================================================================= // FASE B: ELENCO PRODOTTI (XML) // ================================================================= const xmlText = await fileInput.text(); const parser = new XMLParser({ ignoreAttributes: false, textNodeName: "_text", isArray: (name) => name === 'Product', parseTagValue: false }); const parsedData = parser.parse(xmlText); const root = parsedData?.EasyfattProducts || {}; const prodottiXML = root.UpdatedProducts?.Product || root.Products?.Product || []; const arrayProdottiDaSalvare = []; for (const p of prodottiXML) { if (!p.Code) continue; const formatPrice = (val: any) => { if (val === undefined || val === null || val === "") return null; const num = typeof val === 'string' ? parseFloat(val.replace(',', '.')) : val; if (isNaN(num)) return null; return `€ ${num.toFixed(2).replace('.', ',')}`; }; let iva = null; if (p.Vat !== undefined) iva = typeof p.Vat === 'object' ? p.Vat._text : p.Vat; // ESTRAZIONE PESI E MISURE const um = p.Um || p.SizeUm || p.WeightUm || null; const pesoNetto = p.NetWeight || null; const pesoLordo = p.GrossWeight || null; arrayProdottiDaSalvare.push({ "Cod.": String(p.Code).trim(), "Descrizione": p.Description || null, "Categoria": p.Category || null, "Sottocategoria": p.Subcategory || null, "Listino 1 (ivato)": formatPrice(p.GrossPrice1), "Listino 2 (ivato)": formatPrice(p.GrossPrice2), "Listino 3 (ivato)": formatPrice(p.GrossPrice3), "Cod. Iva": iva || null, "Note": p.Notes || null, "Um": um, "Peso Netto": pesoNetto, "Peso Lordo": pesoLordo }); } const { count: prodottiAttualiInDB } = await supabase .from('prodotti') .select('*', { count: 'exact', head: true }); const dbCount = prodottiAttualiInDB || 0; const arrivoCount = arrayProdottiDaSalvare.length; const forzaIncrementale = (arrivoCount < dbCount && dbCount > 0); if (!forzaIncrementale) { if (arrivoCount > 0) { const { data: existingFiles } = await supabase.storage.from('BIrre').list('', { limit: 5000 }); if (existingFiles) { const codiciValidi = new Set(arrayProdottiDaSalvare.map(p => p['Cod.'])); const fotoDaCancellare = existingFiles .filter(f => f.name.endsWith('.jpg') && !codiciValidi.has(f.name.replace('.jpg', ''))) .map(f => f.name); if (fotoDaCancellare.length > 0) { await supabase.storage.from('BIrre').remove(fotoDaCancellare); } } const { error: delErr } = await supabase.from('prodotti').delete().neq('Descrizione', 'xxxxxxxx_dummy_text_xxxxxxxx'); if (delErr) throw delErr; const { error: insErr } = await supabase.from('prodotti').insert(arrayProdottiDaSalvare); if (insErr) throw insErr; } } else { if (arrivoCount > 0) { const { error: upErr } = await supabase.from('prodotti').upsert(arrayProdottiDaSalvare, { onConflict: 'Cod.' }); if (upErr) throw upErr; } } if (root.DeletedProducts?.Product) { const codiciDaCancellare = []; for (const p of root.DeletedProducts.Product) { if (p.Code) codiciDaCancellare.push(String(p.Code).trim()); } if (codiciDaCancellare.length > 0) { await supabase.from('prodotti').delete().in('Cod.', codiciDaCancellare); const fotoDaCancellare = codiciDaCancellare.map(cod => `${cod}.jpg`); await supabase.storage.from('BIrre').remove(fotoDaCancellare); } } const publicUrl = `${supabaseUrl}/functions/v1/danea-sync`; const responseText = `OK\nImageSendURL=${publicUrl}`; return new Response(responseText, { status: 200, headers: { "Content-Type": "text/plain" } }); } catch (err) { console.error("ERRORE DURANTE LA SINCRONIZZAZIONE:", err); return new Response(`Errore: ${err.message}`, { status: 500 }); } })
